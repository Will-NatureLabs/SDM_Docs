[docs]    def read_X_data(self, X_data_folder=None, columns=None):
        """
        Reads the explanatory variable data from a CSV file located in the specified folder.

        This function expects exactly one CSV file in the given folder. It reads the CSV into a pandas DataFrame.
        If the folder contains more or less than one CSV file, an error message is displayed.

        Args:
            X_data_folder (str, optional): The folder path where the CSV file is located. Defaults to self.X_data_folder.
            columns (list of str, optional): A list of column names to read from the CSV file. 
                                            If None, all columns will be read. Defaults to None.

        Returns:
            pd.DataFrame: The DataFrame containing the data read from the CSV file.

        Raises:
            Exception: If the number of CSV files in the folder is not exactly 1.
        """
        if X_data_folder is None:
            X_data_folder = self.X_data_folder

        # List all files in the directory
        files = os.listdir(X_data_folder)

        # Filter the list of files to only include CSV files
        csv_files = [f for f in files if f.endswith('.csv')]

        # Check the number of CSV files in the folder
        if len(csv_files) == 1:
            # Read the CSV file using pandas
            csv_file_path = os.path.join(X_data_folder, csv_files[0])
            if columns is None: 
                df = pd.read_csv(csv_file_path)
            else:
                df = pd.read_csv(csv_file_path, usecols=columns)
            print(f"Successfully read the CSV file from: \n{csv_file_path}", csv_files[0])
        else:
            print("Error: The folder should contain exactly 1 CSV file. Found", len(csv_files), "CSV files.")
            return None
        return df



[docs]    def df_to_gdf_point(self, df, lon_column='lon', lat_column='lat', crs='EPSG:4326', drop_coords=False):
        """
        Converts a pandas DataFrame to a GeoPandas GeoDataFrame, creating point geometries from latitude and longitude columns.

        Args:
            df (pd.DataFrame): The original pandas DataFrame.
            lon_column (str, optional): The name of the longitude column in df. Defaults to 'lon'.
            lat_column (str, optional): The name of the latitude column in df. Defaults to 'lat'.
            crs (str, optional): The coordinate reference system to set for the resulting GeoDataFrame. Defaults to 'EPSG:4326'.
            drop_coords (bool, optional): Whether or not to drop the original latitude and longitude columns. Defaults to False.

        Returns:
            gpd.GeoDataFrame: The converted GeoDataFrame with point geometries.
        """
        # Create a GeoDataFrame with the original DataFrame and the new geometry
        gdf = gpd.GeoDataFrame(df, geometry=gpd.points_from_xy(df[lon_column], df[lat_column]), crs=crs)

        if drop_coords is True:
            gdf.drop(columns=[lon_column, lat_column], inplace=True)
        
        return gdf

    @contextmanager
    def suppress_output():
        """Context manager to suppress standard output, display function, and interactive mode for pyplot.

        This context manager is useful for suppressing print statements, IPython display calls, and
        pyplot plots during the execution of a given code block or function. When the code block
        or function is done, the original behavior is restored.

        Example:
            def example_function():
                print("This print statement will be suppressed.")
                display("This display call will be suppressed.")
                plt.plot([0, 1, 2], [0, 1, 4])
                plt.show()

            with suppress_output():
                example_function()

        Yields:
            None
        """
        # Suppress prints
        orig_stdout = sys.stdout
        sys.stdout = open(os.devnull, 'w')
        
        # Supress iPython display
        global display
        orig_display = display
        display = lambda *args, **kwargs: None
        
        # Suppress plots
        orig_backend = matplotlib.get_backend()
        plt.switch_backend('Agg')  # Set backend to Agg to suppress plots
        plt.ioff()  # Turn off interactive mode for pyplot

        yield  # This is where the provided function will run

        # Restore standard output, display function, interactive mode for pyplot, and backend if they were redirected
        sys.stdout = orig_stdout
        display = orig_display
        plt.ion()  # Turn on interactive mode for pyplot
        plt.switch_backend(orig_backend)  # Restore original backend

[docs]    def reclassify_species(self, df, mapping_path=None, pattern='_isPresent'):
        """
        Reclassifies species data in a DataFrame into higher taxonomic ranks like genus, family, or order.

        The method uses a mapping from species to class (e.g., genus/family/order) read from a CSV file. The original DataFrame should have columns containing species presence data, identified by a pattern in the column name (default is '_isPresent'). The method replaces these species columns with new columns corresponding to the higher-level taxonomic classes and updates the DataFrame accordingly.

        Args:
            df (pd.DataFrame): The original DataFrame containing species data. Should have columns with a specific pattern in the name to be identified.
            mapping_path (str, optional): The path to a CSV file containing the mapping between species and higher-level taxonomic classes. If not provided, the method looks for a 'species_mapping.csv' file in the processing folder.
            pattern (str, optional): The pattern used to identify columns containing species data in the DataFrame. Defaults to '_isPresent'.

        Returns:
            pd.DataFrame: The modified DataFrame with species columns replaced by their higher-level taxonomic classes.

        Raises:
            FileNotFoundError: If the mapping file is not found at the specified or default path.
            KeyError: If a species in the DataFrame is not found in the mapping.
        """
        # If no mapping path is explicity provided, check in the processing_folder for one.
        if mapping_path is None: 
            mapping_path = os.path.join(self.processing_folder, 'species_mapping.csv')

        # Read your CSV file with the species and class mapping
        print(f'Reading the species mapping from {mapping_path}')
        mapping_df = pd.read_csv(mapping_path)

        # Convert species names to lowercase in the mapping dictionary
        mapping_dict = mapping_df.set_index('Species')['Class'].to_dict()
        mapping_dict = {k.lower(): v for k, v in mapping_dict.items()}

        # Identify columns that match the pattern and have a corresponding class in the mapping
        species_cols = [col for col in df.columns if pattern in col]

        # Map each column name (without the pattern, converted to lowercase) to its corresponding class
        class_mapping = {col: mapping_dict[col.replace(f'{pattern}', '').lower()] for col in species_cols if col.replace(f'{pattern}', '').lower() in mapping_dict}

        # Create a new dataframe with the classes as columns
        class_df = pd.DataFrame(index=df.index)

        # Iterate over the matching columns and accumulate the values per class
        for col in species_cols:
            if col in class_mapping:
                cls = class_mapping[col]
                if cls in class_df.columns:
                    class_df[cls] = np.maximum(class_df[cls], df[col])
                else:
                    class_df[cls] = df[col]

        # Remove the matching columns from the original dataframe
        df = df.drop(columns=species_cols)

        # Add the new class columns to the original dataframe
        df = pd.concat([df, class_df], axis=1)

        return df

    

[docs]    def display_tif_files(self, directory, threshold=False):
        # Get a list of all .tif files in the directory
        files = [file for file in os.listdir(directory) if file.endswith(".tif")]

        # Calculate the number of rows and columns for the grid
        num_files = len(files)
        num_cols = int(num_files ** 0.5)
        num_rows = (num_files + num_cols - 1) // num_cols

        # Create the grid of subplots
        fig, axs = plt.subplots(num_rows, num_cols, figsize=(96, 64))

        # Make axs array always 2D for uniformity
        if num_rows == 1 and num_cols == 1:
            axs = np.array([[axs]])
        elif num_rows == 1:
            axs = np.expand_dims(axs, axis=0)
        elif num_cols == 1:
            axs = np.expand_dims(axs, axis=1)

        # Iterate over the files and display each image in a subplot
        for i, file in enumerate(files):
            # Calculate the subplot indices
            row_idx = i // num_cols
            col_idx = i % num_cols

            # Load the image using imageio
            img = imageio.imread(os.path.join(directory, file))

            # Threshold the image if requested
            if threshold:
                img = np.where(img > 0.5, 1, 0)

            # Display the image in the corresponding subplot
            ax = axs[row_idx, col_idx]
            ax.imshow(img)
            species_name = str(file).replace('_isPresent.tif','')
            ax.set_title(species_name, fontsize=50)

            # Remove axis ticks and labels
            ax.axis('off')

        # Adjust the layout and display the plot
        plt.tight_layout()
        plt.show()
