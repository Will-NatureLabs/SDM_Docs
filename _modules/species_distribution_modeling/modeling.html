[docs]    def calculate_alpha(self, df=None):
        """
        Calculates alpha based on species class frequencies in the given DataFrame or the stored y_train.
        This is used to weight the classes in the custom loss functions. 

        Args:
            DataFrame (pandas.DataFrame, optional): A DataFrame containing species class labels. If None, 
                                                   the method will default to self.y_train.
                                                   
        Returns:
            np.array: Array containing the calculated alpha values for each species class.
        """
        if df is None:
            df = self.y_train
        """Calculates alpha based on class frequencies"""
        # Number of instances
        n_instances = df.shape[0]

        # Number of positive instances for each label
        n_positives = np.sum(df, axis=0)

        # Number of negative instances for each label
        n_negatives = n_instances - n_positives

        # Calculate alpha for each label
        self.alpha = n_negatives / (n_positives + n_negatives)

        return self.alpha

[docs]        def __init__(self, gamma=2, alpha=0.5, name="FocalLoss2d", **kwargs):
            """
            Initializes the FocalLoss2d object.
            
            Args:
                gamma (float, optional): The focusing parameter. Default is 2.
                alpha (float, optional): A weighting factor for positive species classes in an imbalanced dataset. Default is 0.5.
                name (str, optional): The name for the loss. Default is "FocalLoss2d".
                **kwargs: Arbitrary keyword arguments passed to parent class.
            """
            super().__init__(name=name, **kwargs)
            self.gamma = float(gamma)
            self.alpha = tf.constant(alpha, dtype=tf.float32)
            self.epsilon = 1.e-7



[docs]        def call(self, y_true, y_pred):
            """
            Computes the Focal Loss between y_true and y_pred for species distribution data.
            
            Args:
                y_true (tf.Tensor): A tensor of true species labels.
                y_pred (tf.Tensor): A tensor of predicted species labels.
                
            Returns:
                tf.Tensor: A tensor representing the Focal Loss value.
            """
            y_true = tf.cast(y_true, tf.float32)
            y_pred = tf.clip_by_value(y_pred, self.epsilon, 1. - self.epsilon)

            alpha_t = y_true*self.alpha + (tf.ones_like(y_true)-y_true)*(1-self.alpha)
            y_t = tf.where(tf.equal(y_true, 1), y_pred, 1.-y_pred)
            ce = -tf.math.log(y_t)
            weight = tf.pow(tf.subtract(1., y_t), self.gamma)
            fl = tf.multiply(tf.multiply(weight, ce), alpha_t)
            loss = tf.reduce_mean(fl)
            return loss

[docs]        def call(self, y_true, y_pred):
            """
            Computes the Weighted Binary Cross-Entropy Loss between y_true and y_pred.

            Args:
                y_true (tf.Tensor): A tensor of true binary labels.
                y_pred (tf.Tensor): A tensor of predicted probabilities from a sigmoid function.

            Returns:
                tf.Tensor: A tensor representing the Weighted Binary Cross-Entropy Loss value.
            """
            # Weights for positive and negative classes for each label
            Wp = 1 - self.alpha
            Wn = self.alpha

            first_term = y_true * K.log(y_pred + K.epsilon())
            second_term = (1 - y_true) * K.log(1 - y_pred + K.epsilon())
            
            # Ensure weights are applied to each label separately
            loss = - (Wp * first_term + Wn * second_term)
            
            return K.mean(loss, axis=-1)

        

[docs]    def hamming_loss(self, y_true, y_pred):
        """
        Calculates the Hamming loss, which measures the fraction of labels that are incorrectly predicted.

        Args:
            y_true (tf.Tensor): A tensor of true labels.
            y_pred (tf.Tensor): A tensor of predicted labels.

        Returns:
            tf.Tensor: A tensor representing the Hamming loss value.
        """
        tmp = K.abs(y_true - y_pred)
        return K.mean(tmp, axis=-1)

        """

                                        If not provided, it defaults to the model_in_folder attribute of the instance.

            ("y_train", "y_train.pkl"),

        for variable_name, file_name in data_files:
            file_path = os.path.join(data_folder, file_name)
            with open(file_path, "rb") as file:
                setattr(self, variable_name, pd.read_pickle(file))
        
        # Set number of classes
        self.num_classes = self.y_train.shape[1] if len(self.y_train.shape) > 1 else 1
        print(
        f"Data loaded successfully. \n"
        f"X_train shape: {self.X_train.shape}, \n"
        f"X_test shape: {self.X_test.shape}, \n"
        f"X_val shape: {self.X_val.shape}, \n"
        f"y_train shape: {self.y_train.shape}, \n"
        f"y_test shape: {self.y_test.shape}, \n"
        f"y_val shape: {self.y_val.shape}"
        )

        self.calculate_alpha()

        return self.X_train, self.X_test, self.X_val, self.y_train, self.y_test, self.y_val

                 relaxation_factor=1.0, sparsity_coefficient=0.06, batch_momentum=.9, 
                 learning_rate=0.002, virtual_batch_size=32, batch_size=64, loss='binary_crossentropy', **kwargs):
        """
        Initializes a new TabNet model with the given parameters.

        Args:
            X_train (array-like, optional): Training data. If not provided, will look for data saved to the model object.
            y_train (array-like, optional): Training labels. If not provided, will look for data saved to the model object.
            feature_dim (int, optional): The dimensionality of the hidden layers in TabNet. Default is 512.
            dim_ratio (float, optional): The ratio for the dimension of output space to feature space. Default is 0.5.
            num_decision_steps (int, optional): The number of decision steps in the TabNet model. Default is 8.
            relaxation_factor (float, optional): The relaxation factor for TabNet's feature selection. Default is 1.0.
            sparsity_coefficient (float, optional): The sparsity penalty. Default is 0.06.
            batch_momentum (float, optional): The momentum term for batch normalization. Default is 0.9.
            learning_rate (float, optional): The learning rate for the optimizer. Default is 0.002.
            virtual_batch_size (int, optional): The virtual batch size for TabNet's Ghost Batch Normalization. Default is 32.
            loss (str, optional): The loss function to use. It could be 'binary_crossentropy', 'focal_loss', or 'wBCE'. Default is 'binary_crossentropy'.
            **kwargs: Arbitrary keyword arguments.

        Returns:
            None

        Notes:
            If custom loss functions 'focal_loss' or 'wBCE' are chosen, the corresponding custom loss classes are utilized.
            The model is initialized with certain regularizations and modifications suited for the multilabel problem and is then compiled.
        """
        # If data isn't directly provided, look for data saved to the model object
        if X_train is None: 
            X_train = self.X_train
        if y_train is None: 
            y_train = self.y_train
        
        # Store hyperparameter data against the class instance
        self.output_dim = int(feature_dim * dim_ratio)
        self.feature_dim = feature_dim
        self.dim_ratio = dim_ratio
        self.num_decision_steps = num_decision_steps
        self.relaxation_factor = relaxation_factor
        self.sparsity_coefficient = sparsity_coefficient
        self.batch_momentum = batch_momentum
        self.learning_rate = learning_rate
        self.virtual_batch_size = virtual_batch_size
        self.batch_size = batch_size

        # set the hyperparameters
        self.hyperparameters = {
        'feature_dim' : self.feature_dim,
        'output_dim' : self.output_dim,
        'num_decision_steps' : self.num_decision_steps,
        'relaxation_factor' : self.relaxation_factor,
        'sparsity_coefficient' : self.sparsity_coefficient,
        'batch_momentum' : self.batch_momentum,
        'learning_rate' : self.learning_rate,
        'virtual_batch_size' : self.virtual_batch_size,
        'batch_size' : self.batch_size
        }

        # Store the selected loss in the class instance
        # Check if custom loss function was selected
        if loss == 'focal_loss':
            self.loss = focal_loss = self.FocalLoss2d(gamma=2, alpha=self.alpha)
        elif loss == 'wBCE': 
            self.loss = focal_loss = self.WeightedBCELoss2d(alpha=self.alpha)
        # If not, use the provided loss function (String or tf.loss object)
        else:
            self.loss = loss

        # Store the total number of classes
        self.num_classes = y_train.shape[1] if len(y_train.shape) > 1 else 1

        # Initialize the TabNetClassifier model adapted from https://github.com/titu1994/tf-TabNet.git
        # Modifications:
        # 1. The model has slight L2 regularisation applied.
        # 2. The activation function is changed from softmax to sigmoid to allow it to generate independant probabilities in a multilabel problem. 
        self.model = TabNetClassifier(feature_columns=None, num_classes=self.num_classes, num_features=X_train.shape[1],
                                    feature_dim=self.feature_dim, output_dim=self.output_dim,
                                    num_decision_steps=self.num_decision_steps, relaxation_factor=self.relaxation_factor,
                                    sparsity_coefficient=self.sparsity_coefficient, batch_momentum=self.batch_momentum,
                                    virtual_batch_size=self.virtual_batch_size, norm_type='group',
                                    num_groups=-1)

        # Compile the model
        lr_schedule = tf.keras.optimizers.schedules.ExponentialDecay(self.learning_rate, decay_steps=500, decay_rate=0.9, staircase=False)
        optimizer = tf.keras.optimizers.Adam(learning_rate=lr_schedule)
        self.model.compile(optimizer=optimizer, loss=self.loss, 
                    metrics=[self.hamming_loss])
        print(f'Model initialised succesfully with: \n Loss function: {self.loss} \n Hyperparameters: \n {self.hyperparameters}')

        """
        Trains the TabNet model with provided data and settings, and logs the training progress and feature masks.

        Args:
            model_name (str, optional): The name of the saved best model. Default is 'best_model'.
            X_train (array-like, optional): Training data. Defaults to the instance's attribute if not provided.
            y_train (array-like, optional): Training labels. Defaults to the instance's attribute if not provided.
            X_val (array-like, optional): Validation data. Defaults to the instance's attribute if not provided.
            y_val (array-like, optional): Validation labels. Defaults to the instance's attribute if not provided.
            output_folder (str, optional): Directory for saving the trained model and logs. Defaults to the instance's attribute if not provided.
            epochs (int, optional): The number of epochs for training. Default is 3.
            patience (int, optional): The number of epochs with no improvement after which training will be stopped. Default is 3.
            batch_size (int, optional): The size of batches for training. Default is 32.

        Returns:
            None

        Notes:
            - This function ensures that the model is initialized before training.
            - If the model is loaded from a saved state, feature masks are not generated.
            - Callbacks such as TensorBoard logging, early stopping, and model checkpoints are used during training.
            - The feature masks are saved to the TensorBoard logs.
        """
        # If data is not provided, use instance's attributes
        if X_train is None:
            X_train = self.X_train
        if y_train is None:
            y_train = self.y_train
        if X_val is None:
            X_val = self.X_val
        if y_val is None:
            y_val = self.y_val
        if output_folder is None:
            output_folder = self.model_out_folder
        try:
            batch_size = self.hyperparameters.get('batch_size')
        except:
            print('No batch size found in the SDM object. \nPlease pass this parameter manually or run hyperparameter optimisation to calculate and assign a batch_size to the SDM object.')

        # Ensure model is initialized
        if self.model is None:
            self.initialize_model(X_train, y_train)
            print(f"Model not yet initialised. \n Initialising using default loss: binary_crossentropy and default hyperparameters: {self.hyperparameters}.")

        # Generate a unique timestamp
        timestamp = datetime.datetime.now().strftime('%Y-%m-%d_%H-%M')

        # Use the timestamp to create a unique directory for the current run 
        self.model_output_folder = os.path.join(output_folder, timestamp)
        if not os.path.exists(self.model_output_folder):
            os.makedirs(self.model_output_folder)
        print(f"Saving model outputs to: {self.model_output_folder}")
        
        # Set log and best model paths
        self.log_folder = os.path.join(self.model_output_folder, 'logs')
        self.best_model_folder = os.path.join(self.model_output_folder, model_name)

        self.log_hyperparameters(self.hyperparameters, self.log_folder)
        csv_logger = keras.callbacks.CSVLogger(os.path.join(self.log_folder, 'tabnet_training.log'))
        checkpoint = keras.callbacks.ModelCheckpoint(self.best_model_folder,
                                                    save_best_only=True, save_weights_only=False, verbose=1, save_format='tf')
        tensorboard_callback = keras.callbacks.TensorBoard(log_dir=self.log_folder, histogram_freq=1)
        early_stopping = EarlyStopping(monitor='val_loss', patience=patience, verbose=1)
        livelossplot_callback = PlotLossesKeras()

        # Train the model
        print(f'Training the model with hyperparameters: {self.hyperparameters} \nand loss function:{self.loss}')
        history = self.model.fit(X_train.values, y_train.values, epochs=epochs, batch_size=batch_size, validation_data=(X_val.values, y_val.values),
        callbacks=[checkpoint, csv_logger, tensorboard_callback, early_stopping, livelossplot_callback])
        
        # Generate feature masks
        # Put the model in eager mode (necessary to access the masks)
        x_sample, _ = next(iter(tf.data.Dataset.from_tensor_slices((X_train.values, y_train.values)).batch(1)))
        _ = self.model(x_sample)

        # Feature masks are not saved when a model is saved, they have to be accessed during the first training run
        if self.loaded_model: 
            print('Cannot generate feature masks from a loaded model, skipping feature mask generation.')
        if not self.loaded_model:
            # Setup TensorBoard writer
            writer = tf.summary.create_file_writer(self.log_folder)
            with writer.as_default():
                for i, mask in enumerate(self.model.tabnet.feature_selection_masks):
                    print("Saving mask {} of shape {}".format(i + 1, mask.shape))
                    tf.summary.image('Mask at decision step (layer) {}'.format(i + 1), step=0, data=mask, max_outputs=1)
                    writer.flush()
                agg_mask = self.model.tabnet.aggregate_feature_selection_mask
                print("Saving aggregate mask of shape", agg_mask.shape)
                tf.summary.image("Aggregate Mask", step=0, data=agg_mask, max_outputs=1)
                writer.flush()
            writer.close()
            self.feature_mask = self.model.tabnet.aggregate_feature_selection_mask
            return history, self.feature_mask, self.model_output_folder
        else:
            return history, self.model_output_folder

                - y_pred_labels (array-like): Predicted labels (binary classification) for each class based on the threshold.

        Notes:
            - For multilabel classification, the best threshold for each label will be computed separately.
        """
        # if a threshold is specified, calculate_best_thresholds is set to false, otherwise best thresholds will be used instead. 
        if threshold is not None: 
            calculate_best_thresholds = False

        # Predict on the test data by default
        if X is None: 
            X = self.X_test
        if y is None: 
            y = self.y_test

        # Handle single labels
        if self.num_classes == 1:
            self.y_pred_prob = self.model.predict(X)
        elif self.num_classes > 1:
            self.y_pred_prob = self.model.predict(X.values)
            
        if calculate_best_thresholds is True:
            self.calculate_best_thresholds()

            # Threshold probability predictions to create labels based on the threshold that yields the highest F1 score in the validation data.
            self.y_pred_labels = self.apply_thresholds(self.y_pred_prob, self.best_thresholds)

        else:
            # Threshold probability predictions to create labels based on the user defined threshold. 
            self.y_pred_labels = tf.cast(self.y_pred_prob > threshold, tf.int32)
        return self.y_pred_prob, self.y_pred_labels



[docs]    def calculate_best_thresholds(self, X=None, y=None, metric_to_maximize="f1"):
        """
        Calculate the best decision thresholds for each class based on a specific metric.

        The function uses the precision-recall curve to find a decision threshold for each class
        that maximizes a given metric - F1 score, precision, or recall. If both precision and 
        recall for a class are zero, a default threshold of 0.5 is assigned.

        Args:
            X (pd.DataFrame, optional): Features. If not provided, self.X_val will be used.
            y (pd.DataFrame, optional): True labels. If not provided, self.y_val will be used.
            metric_to_maximize (str, optional): Metric to be maximized when choosing the best threshold.
                Must be one of ["f1", "precision", "recall"]. Default is "f1".

        Raises:
            AssertionError: If metric_to_maximize is not one of ["f1", "precision", "recall"].

        Returns:
            tuple: A tuple containing two elements:
                - list of float: Best decision thresholds for each class.
                - np.array: An array with the same shape as threshold_y_pred_prob initialized to zeros.
        """        
        if X is None: 
            X = self.X_val
        if y is None: 
            y = self.y_val

        y_pred_prob = self.model.predict(X)

        # This process requires the y_true to be binary
        y_true = (y > .5).astype(int)  

        assert metric_to_maximize in ["f1", "precision", "recall"], 'metric_to_maximize must be either "f1", "precision", or "recall"'

        best_thresholds = []
        y_pred_labels = np.zeros(y_pred_prob.shape)

        for i in range(self.num_classes):
            precision, recall, thresholds = precision_recall_curve(y_true.iloc[:, i], y_pred_prob[:, i])
            precision = precision[:-1]
            recall = recall[:-1]
            thresholds = np.append(thresholds, 1)

            if precision.size != 0 and recall.size != 0:  # Precision and recall are not both zero
                if metric_to_maximize == "f1":
                    scores = 2*(precision*recall)/(precision+recall+1e-10)  # Avoid division by zero
                elif metric_to_maximize == "precision":
                    scores = precision
                elif metric_to_maximize == "recall":
                    scores = recall

                best_threshold = thresholds[np.argmax(scores)]
            else:  # Precision and recall are both zero
                best_threshold = 0.5

            best_thresholds.append(best_threshold)

        self.best_thresholds = best_thresholds
        return self.best_thresholds, y_pred_labels 

    

[docs]    def apply_thresholds(self, y_pred_prob, thresholds):
        """
        Apply the specified thresholds to the predicted probabilities.

        For each class, labels are assigned based on whether the predicted probability 
        for that class is greater than the corresponding threshold.

        Args:
            y_pred_prob (np.array): 2D array of predicted probabilities for each class.
            thresholds (list): Thresholds for each class.

        Returns:
            np.array: 2D array of predicted labels for each class.
        """
        y_pred_labels = np.zeros(y_pred_prob.shape)

        for i in range(self.num_classes):
            y_pred_labels[:, i] = (y_pred_prob[:, i] > thresholds[i]).astype(int)

        return y_pred_labels

            precision_labelwise = precision_score(y_true, y_pred_labels, average=None, zero_division=1)
            recall_labelwise = recall_score(y_true, y_pred_labels, average=None, zero_division=1)

            if all(len(np.unique(y_true[:, i])) > 1 for i in range(y_true.shape[1])):
                roc_auc_labelwise = roc_auc_score(y_true, y_pred_prob)

            g = sns.catplot(x='Value', y='Metric', col='Species', col_wrap=col_wrap, data=labelwise_melted, kind='bar', height=4, aspect=0.8, palette="viridis_r")

        return summary_metrics, labelwise_metrics

    def probability_adjusted_simpsons(probabilities):
        """
        Calculate the Simpson's biodiversity Index.
        This adjusted index will range from 0 to (1-1/s), where s is the number of species (or categories). 
        A higher value indicates greater biodiversity or category diversity. 
        This is a common transformation of Simpson's Index used to make the scale more intuitive, where higher values reflect higher diversity.

        This quantifies diversity in a way that accounts for both richness (number of categories) 
        and evenness (how equally members are distributed among categories).
        
        Args:
            probabilities (np.array or pd.Series): Probabilities for the distinct categories.
            
        Returns:
            float: Probability-Adjusted Simpson's Index.
        """    
        return 1-np.sum(probabilities**2)

    def shannon_diversity(probabilities):
        """
        Calculate the Shannon Diversity Index based on a row of probabilities.
        
        Note: This method is an alternative biodiversity index. It's not used by default 
        as it relies on balanced classes.

        The reason for this caveat is that the Shannon index can be sensitive to rare categories, 
        which could skew the measure of diversity if classes are highly imbalanced, which is likely for species observation data.

        Args:
            probabilities (pd.Series): Probabilities for the distinct categories.
            
        Returns:
            float: Shannon Diversity Index.
        """
        probabilities = probabilities.dropna()  # Drop NaN values, if any
        shannon_index = -np.sum(probabilities * np.log(probabilities))
        return shannon_index

[docs]    def predict_and_plot_raster(self, raster_folder, processing_folder=None):
        """
        Reads raster data from a folder, scales the features, makes predictions using a trained model,
        and creates raster output files for each species as well as a species richness raster.
        Note that unlike the sister function predict_and_plot_raster, this function does not take a lon/lat column.
        This is because the coordinates are instead calculated from the centroids of each cell in the raster grid. 

        Args:
            raster_folder (str): Path to the folder containing the raster files.
            processing_folder (str, optional): Directory to find models and scalers. Defaults to the object's processing_folder attribute.

        Side Effects:
            - Creates and saves raster files for each species prediction.
            - Creates and saves a species richness raster.
            - Prints status messages regarding the scaling and exporting of files.

        Raises:
            AssertionError: If the rasters do not have the same geotransform, CRS, or shape.

        Notes: 
            Please ensure the raster files all share the same extent and geotransform (this is handled automatically by the interplate_to_raster function).
            Please also ensure that the rasters are in the same CRS as the model's training data. 

        """
        combined_data = pd.DataFrame()
        geotransform = None
        shape = None
        crs = None
        lon = None
        lat = None
        raster_filepaths = []
        all_rasters = []
        model = self.model
        if processing_folder is None:
            processing_folder = self.processing_folder

        for filename in os.listdir(raster_folder):
            if filename.endswith(('.tif', '.tiff')):
                raster_path = os.path.join(raster_folder, filename)
                raster_filepaths.append(raster_path)

        for raster_path in raster_filepaths:
            # Read the raster grid file
            with rasterio.open(raster_path) as src:
                raster = src.read(1)  # Assuming the data is stored in the first band

                # Check the rasters have the same geotransform, shape and crs, save these as variables. 
                if not geotransform:
                    geotransform = src.transform
                else:
                    assert geotransform.almost_equals(src.transform)

                if not crs:
                    crs = src.crs
                else:
                    assert crs == src.crs

                if not shape:
                    shape = raster.shape
                else:
                    assert shape == raster.shape

            # Flatten the grid data and add it as a column to the combined DataFrame
            flattened_raster = raster.flatten()

            # use the filename as the variable name
            column_name = os.path.splitext(os.path.basename(raster_path))[0]
            combined_data[column_name] = flattened_raster
            all_rasters.append(raster)
            
        # Get species names
        if isinstance(self.y_train, pd.DataFrame):  # If y_train is a DataFrame
            unique_species_list = self.y_train.columns.tolist()  # Convert DataFrame column names to list
        elif isinstance(self.y_train, pd.Series):  # If y_train is a Series (single label)
            unique_species_list = [self.y_train.name]  # Get the name of the series
            
        # calculate lat/lon of raster cell centroids
        with rasterio.open(raster_filepaths[0]) as src:
            raster = src.read(1)  # Assuming the grid data is stored in the first band

        # Filter out nan values (sometimes caused by masking), get pixel coordinates
        rows, cols = np.where(raster != src.nodata)

        # Convert pixel coordinates to longitude and latitude
        lon, lat = rasterio.transform.xy(geotransform, rows, cols)
        lon = pd.Series(lon, name='lon', dtype='float64')
        lat = pd.Series(lat, name='lat', dtype='float64')
        print(f'Calculated coordinate from the centroids of each raster cell: \n')
      
        # Load the coord_scaler scaler from the file
        print('Loading the coord_scaler and the numeric_variable_scaler from the processing folder')
        with open(os.path.join(processing_folder, 'coord_scaler.pkl'), 'rb') as f:
            coord_scaler = pickle.load(f)   

        # Load the numeric_variable_scaler.pkl scaler from the file
        with open(os.path.join(processing_folder, 'numeric_variable_scaler.pkl'), 'rb') as f:
            variable_scaler = pickle.load(f)  

        # Join the centroid coords to the dataframe
        combined_data = pd.concat([lon, lat, combined_data], axis=1)

        print(f'Adding centroid coordinates to the dataframe. \nData example: \n')

        # Select the variable columns
        variable_cols = combined_data.drop(columns=['lon', 'lat']).columns.to_list()
        sorted_variable_cols = sorted(variable_cols)

        self.utils.s_print(f"Column, Datatype for variable columns: {[(column, datatype) for column, datatype in zip(sorted_variable_cols, combined_data[sorted_variable_cols].dtypes.tolist())]}")

        print('\n**These must match those printed by preprocess_X_data as we are re-using the same scaler.**\n')
        # Apply the scaler to the flattened data
        print('Applying the coord_scaler and numeric_variable_scaler to the GeoDataFrame')
        combined_data[['lon', 'lat']] = coord_scaler.transform(combined_data[['lon', 'lat']])
        print('Coordinate scaling complete')

        combined_data[sorted_variable_cols] = variable_scaler.transform(combined_data[sorted_variable_cols])
        print('Variable scaling complete')

        # Convert to np array for predictions
        X = np.array(combined_data)

        # Make predictions for all cells using the model    
        predictions = model.predict(X)
        
        # Reshape predictions to match the output shape
        predictions = np.reshape(predictions, (shape[0], shape[1], self.y_train.shape[1]))

        # Create separate raster layers for each column in the prediction
        for i, col_name in enumerate(unique_species_list): 
            output_filename = os.path.join(self.model_output_folder, 'predicted_rasters', f'{col_name}.tif')
            
            # Create the directory if it doesn't exist
            output_directory = os.path.dirname(output_filename)
            os.makedirs(output_directory, exist_ok=True)

            # Create a new raster file
            output_raster = rasterio.open(output_filename, 'w', driver='GTiff', height=shape[0], width=shape[1],
                                        count=1, dtype=predictions.dtype, crs=crs, transform=geotransform)

            # Write the prediction values to the raster file
            output_raster.write(predictions[:, :, i], 1)

            # Close the raster file
            output_raster.close()
            print(f'Exported {output_filename}')

        # Calculate species richness raster
        species_richness_raster = np.sum(all_rasters, axis=0)

        # Create output filename for the species_richness raster
        predicted_raster_folder = os.path.join(self.model_output_folder, 'predicted_rasters')
        species_richness_filename = os.path.join(predicted_raster_folder, 'species_richness.tif')

        # Create the directory if it doesn't exist
        output_directory = os.path.dirname(species_richness_filename)
        os.makedirs(output_directory, exist_ok=True)

        # Create a new raster file for the species_richness raster
        species_richness_output_raster = rasterio.open(species_richness_filename, 'w', driver='GTiff', height=shape[0], width=shape[1],
                                                count=1, dtype=species_richness_raster.dtype, crs=crs, transform=geotransform)

        # Write the species_richness values to the raster file
        species_richness_output_raster.write(species_richness_raster, 1)

        # Close the species_richness raster file
        species_richness_output_raster.close()
        print(f'Exported {species_richness_filename}')
       
        self.utils.display_tif_files(predicted_raster_folder)

        """
        Optimize the hyperparameters for a machine learning model using Optuna.

        Args:
            n_trials (int): Number of trials for hyperparameter optimization.
            epochs_per_trial (int, optional): Number of epochs per trial. Defaults to 20.
            loss (str, optional): Loss function to use. Defaults to 'binary_crossentropy'.
            X_train (DataFrame, optional): Training features. Defaults to None.
            y_train (DataFrame, optional): Training labels. Defaults to None.
            X_val (DataFrame, optional): Validation features. Defaults to None.
            y_val (DataFrame, optional): Validation labels. Defaults to None.
            model_out_folder (str, optional): Folder to save optimization study and best parameters. Defaults to None.

        Returns:
            tuple: Best hyperparameters and the Optuna study object.

        Raises:
            FileNotFoundError: If 'model_out_folder' does not exist and is not None.
            ValueError: If the training and validation data shapes do not match.
            Exception: For other types of errors during optimization.

        Note:
            The function also saves plots related to the optimization in the 'model_out_folder'.
        """
        # User can either pass dataframes directly or use the load_data function first to assign the data to class variables, 
        # then run the function without passing datasets to it explicitly.
        if X_train is None:
            X_train = self.X_train
        if y_train is None:
            y_train = self.y_train
        if X_val is None:
            X_val = self.X_val
        if y_val is None:
            y_val = self.y_val
        if model_out_folder is None:
            model_out_folder = self.model_out_folder
        self.loss = loss

        hyperoptim_outpath = os.path.join(model_out_folder, 'Hyperparameter Optimization')

        # Define the study's output file path 
        study_file_path = os.path.join(hyperoptim_outpath , 'Study.pkl')
        # Check if the file exists
        if os.path.exists(study_file_path):
            # Load existing study
            print("Loading existing study...")

                study = pickle.load(f)

            # Create a new study
            print("Creating new study...")
            study = optuna.create_study(direction='minimize')

        # Check if the file exists
        if os.path.exists(study_file_path):
            # Load existing study
            print("Loading existing study and continuing hyperparameter optimisation")

        def objective(trial):
            # Hyperparameters to optimise 
            # Hyperparameter suggestions based on advice from the model's authors: https://github.com/titu1994/tf-TabNet/blob/master/tabnet/tabnet.py
            # as well as this article: https://syslog.ravelin.com/classification-with-tabnet-deep-dive-49a0dcc8f7e8
            feature_dim = trial.suggest_int('feature_dim', 32, 512, 32)
            dim_ratio = trial.suggest_float('dim_ratio', 0.1, 0.99, step=0.1)  # updated here
            output_dim = int(feature_dim * dim_ratio)
            num_decision_steps = trial.suggest_int('num_decision_steps', 3, 10, 1)
            relaxation_factor = trial.suggest_float('relaxation_factor', 1, 3.0, step=0.1)  # updated here
            sparsity_coefficient = trial.suggest_float('sparsity_coefficient', 0.0, 0.1, step=0.01)  # updated here
            batch_momentum = trial.suggest_float('batch_momentum', 0.0, 1.0, step=0.1)  # updated here
            learning_rate = trial.suggest_loguniform('learning_rate', 0.001, 0.1)  # logarithmically sample
            virtual_batch_size = trial.suggest_categorical('virtual_batch_size', [32, 64, 128, 256])
            batch_size = trial.suggest_categorical('batch_size', [64, 128, 256, 512])

            # Construct a model with the suggested hyperparameters
            model = TabNetClassifier(feature_columns=None, num_classes=self.num_classes, num_features=self.X_train.shape[1],

                                    virtual_batch_size=virtual_batch_size, norm_type='group', num_groups=-1)

                        metrics=[self.hamming_loss])

                                validation_data=(X_val.values, y_val.values),
                                callbacks=[TFKerasPruningCallback(trial, 'val_loss')])

            val_loss = history.history['val_loss'][-1]
