[docs]        def get_image(self, images=None, filter_geometry=None):
            if isinstance(images, list) and len(images) > 0:
                # Initialize an empty Image for band concatenation
                combined_image = ee.Image(images[0])
                
                # Start from the second image (if any) and add its bands to the combined_image
                for img_url in images[1:]:
                    img = ee.Image(img_url)
                    combined_image = combined_image.addBands(img)
            else:
                # If a single image URL is provided, or the list is empty, process it as before
                combined_image = ee.Image(images) if images is not None else None

            self.image = combined_image

            if filter_geometry is not None and self.image is not None:
                self.filter_geometry_centroid = filter_geometry.geometry.centroid

                # Check CRS and convert to EPSG:4326 if necessary
                original_crs = filter_geometry.crs
                if original_crs != 'epsg:4326':
                    filter_geometry = filter_geometry.to_crs(epsg=4326)
                self.filter_geometry = geemap.geopandas_to_ee(filter_geometry)
                        
                # Determine geometry type of the first geometry in the GeoDataFrame
                first_geom_type = filter_geometry.geometry.iloc[0].geom_type
                if first_geom_type in ['Polygon', 'MultiPolygon']:
                    self.image = self.image.clip(self.filter_geometry)
                else: 
                    raise ValueError("filter_geometry must be polygon.")
            return self.image

    

[docs]        def get_collection(self, collection_id, start_date=None, end_date=None, filter_geometry=None, aggregation_method='mean'):
            """
            Fetches and aggregates a collection of images from Google Earth Engine over a given date range,
            spatially filters the collection to the specified geometry, and applies an aggregation method.
            Defaults to using the past 2 years of data if start_date and end_date are not provided.

            Args:
                collection_id (str): The ID of the Earth Engine image collection to use.
                start_date (str, optional): The start date for the image collection filter (inclusive), formatted as 'YYYY-MM-DD'.
                    Defaults to two years before the current date if not provided.
                end_date (str, optional): The end date for the image collection filter (inclusive), formatted as 'YYYY-MM-DD'.
                    Defaults to the current date if not provided.
                filter_geometry (geopandas.GeoDataFrame, optional): A GeoDataFrame specifying the spatial extent to filter the collection. Defaults to None.
                aggregation_method (str, optional): The method used to aggregate images over time. Defaults to 'mean'.
                    Supported methods: 'mean', 'median', 'min', 'max', 'sum', 'mosaic'.

            Returns:
                ee.Image: An Earth Engine image object resulting from the aggregated collection.
            """
            # Calculate default dates if not provided
            if end_date is None:
                end_date = datetime.utcnow().strftime('%Y-%m-%d')  # Today's date
            if start_date is None:
                start_date = (datetime.utcnow() - timedelta(days=2*365)).strftime('%Y-%m-%d')  # Two years before today

            # Load the image collection
            collection = ee.ImageCollection(collection_id)

            # Filter the collection by date
            collection = collection.filterDate(start_date, end_date)

            # Apply the spatial filter if provided
            if filter_geometry is not None:
                # Check CRS and convert to EPSG:4326 if necessary
                original_crs = filter_geometry.crs
                if original_crs != 'epsg:4326':
                    filter_geometry = filter_geometry.to_crs(epsg=4326)
                self.filter_geometry_centroid = filter_geometry.geometry.centroid
                self.filter_geometry = geemap.geopandas_to_ee(filter_geometry)
                collection = collection.filterBounds(self.filter_geometry)

            # Aggregate the collection based on the specified method
            if aggregation_method == 'mean':
                aggregated_image = collection.mean()
            elif aggregation_method == 'median':
                aggregated_image = collection.median()
            elif aggregation_method == 'min':
                aggregated_image = collection.min()
            elif aggregation_method == 'max':
                aggregated_image = collection.max()
            elif aggregation_method == 'sum':
                aggregated_image = collection.sum()
            elif aggregation_method == 'mosaic':
                aggregated_image = collection.mosaic()
            else:
                raise ValueError("Unsupported aggregation method. Choose from 'mean', 'median', 'min', 'max', 'sum', 'mosaic'.")

            # If a spatial filter is applied, clip the aggregated image to the geometry
            if filter_geometry is not None:
                aggregated_image = aggregated_image.clip(self.filter_geometry)

            self.image = aggregated_image
            return self.image

        

[docs]        def calculate_ndvi(self, nir_band='B5', red_band='B4', plot=False):
            """
            Calculates the Normalized Difference Vegetation Index (NDVI) for the image stored in this instance.
            
            Args:
                nir_band (str): The name of the Near Infrared (NIR) band in the image.
                red_band (str): The name of the Red band in the image.
            
            Returns:
                self with NDVI image: The instance itself with the NDVI image stored, accessible via self.image.
            """
            # Ensure there is an image to calculate NDVI on
            if self.image is None:
                raise ValueError("No image has been loaded. Use get_image or get_collection to load an image first.")
            
            # Calculate NDVI
            ndvi_image = self.image.normalizedDifference([nir_band, red_band])
            
            if plot == True:
                self.plot_raster(filter_band_names=[nir_band, red_band])
            return ndvi_image



[docs]        def extract_values_for_points(self, point_data):
            """Extracts values for points from the Earth Engine image.

            Args:
                point_data (GeoDataFrame): A GeoDataFrame containing point geometries.

            Returns:
                GeoDataFrame: A GeoDataFrame with the extracted values.
            """

            print(f"Ensuring point data is in WGS 84 (EPSG:4326).")
            point_data = point_data.to_crs(epsg=4326)

            # Convert GeoDataFrame points to Earth Engine feature collection
            ee_features = geemap.geopandas_to_ee(point_data, geodesic=False)

            # Sample the image at the points
            sampled = self.image.sampleRegions(collection=ee_features, scale=self.get_scale().getInfo(), geometries=True)

            # Get information from the sampled feature collection
            info = sampled.getInfo()

            # Process the information to create a DataFrame
            data = []
            for feature in info['features']:
                properties = feature['properties']
                geometry = feature.get('geometry')  # Use .get to avoid TypeError if geometry is None
                if geometry:  # Check if geometry is not None
                    coordinates = geometry['coordinates']
                    data.append({**properties, 'longitude': coordinates[0], 'latitude': coordinates[1]})
                else:
                    # Handle the case where geometry is None
                    print("Warning: A feature's geometry is None and will be skipped.")

            df = pd.DataFrame(data)

            if not df.empty:
                # Convert the DataFrame to a GeoDataFrame
                gdf = gpd.GeoDataFrame(df, geometry=gpd.points_from_xy(df.longitude, df.latitude))
                self.point_gdf = gdf
                return gdf
            else:
                print('Geodataframe is empty.')
                return None

            """
            Plots a GeoDataFrame of points on a Folium map with a Google Satellite Hybrid basemap.

            Parameters:
            - gdf: GeoDataFrame containing the points to plot.
            - initial_location: Tuple of (lat, lon) for the initial map center. If None, the center of the GDF points is used.
            """

            # Calculate the mean of the points for centering the map
            center = self.point_gdf.geometry.centroid.unary_union.centroid
            initial_location = (center.y, center.x)

            # Create the map
            Map = folium.Map(location=initial_location, zoom_start=12)

            # Add Basemap
            basemaps = {'Google Satellite Hybrid': folium.TileLayer(
                tiles='https://mt1.google.com/vt/lyrs=s&x={x}&y={y}&z={z}',
                attr='Google',
                name='Google Satellite Hybrid',
                overlay=True,
                control=True,
            )}
            basemaps['Google Satellite Hybrid'].add_to(Map)

            # Plot each point from the GeoDataFrame
            for _, row in self.point_gdf.iterrows():
                folium.Marker(
                    location=[row.geometry.y, row.geometry.x],
                ).add_to(Map)

            # Optionally: Map.save('map.html') to save the map to an HTML file
            return Map

        

[docs]        def plot_raster(self, filter_band_names=None):
            # Calculate the centroid of the filter_geometry
            if self.filter_geometry is not None:
                # Assuming filter_geometry is a GeoDataFrame
                centroid = self.filter_geometry_centroid
                map_center = [centroid.y, centroid.x]
            else:
                # Default to some global center if no filter_geometry is provided
                map_center = [0, 0]  

            # Create the folium map centered on the filter_geometry's centroid
            Map = folium.Map(location=map_center, zoom_start=8)  # Adjust zoom_start as needed for your use case

            # Add Basemap
            basemaps = {'Google Satellite Hybrid': folium.TileLayer(
                tiles='https://mt1.google.com/vt/lyrs=s&x={x}&y={y}&z={z}',
                attr='Google',
                name='Google Satellite Hybrid',
                overlay=True,
                control=True,
            )}
            basemaps['Google Satellite Hybrid'].add_to(Map)

            # Define a method to add EE image tiles to the map
            def addLayer(self, ee_image_object, vis_params, name):
                map_id_dict = ee.Image(ee_image_object).getMapId(vis_params)
                folium.raster_layers.TileLayer(
                    tiles=map_id_dict['tile_fetcher'].url_format,
                    attr='Map Data &copy; <a href="https://earthengine.google.com/">Google Earth Engine</a>',
                    name=name,
                    overlay=True,
                    control=True
                ).add_to(self)

            # Add EE drawing method to folium
            folium.Map.addLayer = addLayer

            # Display each band as a layer
            band_names = self.get_band_names()
            if filter_band_names is not None:
                band_names = [band for band in band_names if band in filter_band_names]
            for name in band_names:
                band = self.image.select(name)
                stats = band.reduceRegion(
                reducer=ee.Reducer.minMax(),
                geometry=self.filter_geometry,
                scale=30,
                maxPixels=1e9)
                
                # Extract the min and max values.
                min_value = stats.get(name + '_min').getInfo()
                max_value = stats.get(name + '_max').getInfo()

                visParams = {
                'min': min_value,
                'max': max_value,
                'palette': ['blue', 'red']
                }

                Map.addLayer(band, visParams, name)

            Map.add_child(folium.LayerControl())
            display(Map)
            return Map

            # Query the GBIF registry to get the dataset key based on the dataset name
            dataset_search_results = registry.dataset_search(q=dataset_name, limit=1)

            # Extract the dataset key if a result is found
            if dataset_search_results['count'] > 0:
                return dataset_search_results['results'][0]['key']
            else:
                raise ValueError(f"Dataset '{dataset_name}' not found.")

            

[docs]        def get_dataset_name(self, dataset_key):
            # GBIF API endpoint for dataset metadata
            api_url = f"https://api.gbif.org/v1/dataset/{dataset_key}"

            try:
                # Send a GET request to the GBIF API
                response = requests.get(api_url)

                # Check if the request was successful (status code 200)
                if response.status_code == 200:
                    # Parse the JSON response
                    dataset_info = response.json()

                    # Extract the dataset name
                    dataset_name = dataset_info.get('title', 'Dataset name not found')

                    return dataset_name
                else:
                    print(f"Error: {response.status_code} - {response.text}")
            except Exception as e:
                print(f"An error occurred: {e}")



[docs]        def download_occurrence_data(self, search=None, geometry=None, country=None, continent=None, basisOfRecord=None, start_date=None, end_date=None, licence=None, datasetKey=None, datasetName=None, map_results=False, export=False):
            # Build the query parameters
            query_params = {}

            if search:
                query_params['q'] = search

            if geometry:
                query_params['geometry'] = geometry

            if country:
                query_params['country'] = country

            if continent:
                query_params['continent'] = continent

            if basisOfRecord:
                query_params['basisOfRecord'] = basisOfRecord

            if start_date:
                query_params['eventDate[gte]'] = start_date

            if end_date:
                query_params['eventDate[lte]'] = end_date

            if datasetKey:
                query_params['datasetKey'] = datasetKey

            if datasetName:
                datasetKey = self.get_dataset_key(datasetName)
                query_params['datasetKey'] = datasetKey
            
            if licence:
                query_params['licence'] = licence

            # Perform the search
            search_results = occ.search(**query_params, spellCheck=True, hasGeospatialIssue=False, hasCoordinate=True)

            # Extract relevant information from the search results
            occurrence_data = []
            for result in search_results['results']:
                occurrence = {
                    'species_name': result.get('species'),
                    'Basis of record': result.get('basisOfRecord'),
                    'latitude': result.get('decimalLatitude'),
                    'longitude': result.get('decimalLongitude'),
                    'date_recorded': result.get('eventDate'),
                    'Dataset Name': result.get('datasetKey'),
                    'License': result.get('license')

                    # Add more fields as needed
                }
                occurrence_data.append(occurrence)

            # Convert the list of dictionaries to a Pandas DataFrame
            df = pd.DataFrame(occurrence_data)

            # Fetch dataset names in bulk
            dataset_keys = df['Dataset Name'].unique()
            dataset_name_mapping = {key: self.get_dataset_name(key) for key in dataset_keys}

            # Replace dataset keys with names in the DataFrame
            df['Dataset Name'] = df['Dataset Name'].map(dataset_name_mapping)

            # Plot the data on a Folium map
            map_center = [df['latitude'].mean(), df['longitude'].mean()]  # Center the map on the mean coordinates

            if map_results == True:
                folium_map = folium.Map(location=map_center, zoom_start=5)  # You can adjust the zoom level as needed

                for index, row in df.iterrows():
                    folium.Marker(
                        location=[row['latitude'], row['longitude']],
                        popup=row['species_name'],
                    ).add_to(folium_map)

                # Save the map as an HTML file or display it
                folium_map.save('occurrence_map.html')  # You can change the file name if needed

                # Create a GeoDataFrame from the DataFrame
                geometry = [Point(lon, lat) for lon, lat in zip(df['longitude'], df['latitude'])]
                crs = {'init': 'epsg:4326'}  # WGS84 coordinate system

                gdf = gpd.GeoDataFrame(df, geometry=geometry, crs=crs)

                return gdf, folium_map
            else:
                return gdf
