[docs]    def create_species_map(self, species=None, time_column='Day', lon_column='lon', lat_column='lat', summarize=False, endure=True, verbose=True, pattern='_isPresent', download=False):
        """
        Creates a map visualization of the presence of different species over time. 
        Gives insight into the survey pattern and the recordings of the raw data.

        Args:
            species (list, optional): List of species to filter by, case insensitive and returns partial matches. If None, all species are included.
            time_column (str, optional): Column name in DataFrame representing time. Defaults to 'Day'.
            lon_column (str, optional): Column name in DataFrame representing longitude. Defaults to 'lon'.
            lat_column (str, optional): Column name in DataFrame representing latitude. Defaults to 'lat'.
            summarize (bool, optional): If True, summarizes the species presence data before plotting. Defaults to False.
            endure (bool, optional): If True, no duration is set for the animation. If False, a duration of 'P1D' is set. Defaults to True.
            verbose (bool, optional): If True, prints debug information. Defaults to True.
            pattern (str, optional): String pattern used to identify columns containing species presence information. Defaults to '_isPresent'.
            download (bool, optional): If True, the generated map will be downloaded as an HTML file. Defaults to False.

        Returns:
            folium.Map: A Folium map object with the presence of different species visualized.

        Example:
            >>> create_species_map(species=['Eleutherodactylus_coqui', 'Margarops_fuscatus'], download=True)
        """
        # Suppress all outputs if verbose is False
        with self.utils.suppress_output() if not verbose else nullcontext():
            proc = processing.Process(X_data_folder=self.X_data_folder,
                                    y_data_folder=self.y_data_folder,
                                    processing_folder=self.processing_folder,
                                    model_in_folder=self.model_in_folder, # Processing_folder/model_in_folder not needed for preprocess_y_data so passing dummy value
                                    model_out_folder=self.model_out_folder,
                                    utils = self.utils) 
            self.utils.s_print("Preparing X datas")
            
            X_data = self.utils.read_X_data(self.X_data_folder)     
            X_data = X_data.rename(columns={'NAME': 'site'})

            self.utils.s_print("Preparing y data")
            y_data = proc.preprocess_y_data(species_list=species, del_time=False, verbose=verbose)

            self.utils.s_print("Joining X data to y data")
            merged_df = proc.join_X_to_y(X_data, y_data)

            print('Define dimensions')
            dimensions = merged_df[[time_column, lon_column, lat_column]]

            print('xy split')
            _, species = proc.xy_split(merged_df)

            print('concat')
            df = pd.concat([species, dimensions], axis=1)
            self.utils.s_print(f"Filtering data to columns: {df.columns}")

            if summarize is True: 
                print(f'summarise is true. df before')
                display(df)
                df = self.utils.reclassify_species(df, pattern=pattern)  
                print('df after')   
                display(df)

            df.rename(columns=lambda col: col.replace(pattern, ''), inplace=True)
            df = df.sort_values(time_column)
    
            map = folium.Map(location=[df[lat_column].mean(), df[lon_column].mean()], zoom_start=10)

            unique_species = [col for col in df.columns if col not in [time_column, lon_column, lat_column]]
            
            # Assign colors to species in a deterministic order
            species_colors = {species: sns.color_palette('husl', n_colors=len(unique_species))[i] for i, species in enumerate(unique_species)}

            # Convert colors to hex format
            species_colors = {species: mcolors.rgb2hex(color) for species, color in species_colors.items()}
    
            geojson = {'type': 'FeatureCollection', 'features': []}

            for _, row in df.iterrows():
                for species in unique_species:
                    if row[species] == 1:
                        feature = {
                            'type': 'Feature',
                            'geometry': {
                                'type': 'Point',
                                'coordinates': [row[lon_column], row[lat_column]]
                            },
                            'properties': {
                                'time': row['Day'].strftime('%Y-%m-%d %H:%M:%S'),
                                'style': {
                                    'color': ''
                                },
                                'icon': 'circle',
                                'iconstyle': {
                                    'fillColor': species_colors[species],
                                    'fillOpacity': 0.5,
                                    'stroke': 'true',
                                    'radius': 3
                                }
                            }
                        }
                        geojson['features'].append(feature)

            if endure:
                duration=None
            else: 
                duration='P1D'

            TimestampedGeoJson(geojson, min_speed=60, max_speed=200, transition_time=0, duration=duration, date_options='DD-MM-YYYY', time_slider_drag_update=True).add_to(map)

            folium.TileLayer('cartodbpositron').add_to(map)

            legend_html = '''
                <div style="position: fixed; bottom: 50px; left: 50px; z-index:9999; font-size: 14px; background-color: white;
                    padding: 10px; border-radius: 5px; box-shadow: 0 0 5px rgba(0,0,0,0.2);">
                    <h4>Legend</h4>
            '''
            for species, color in species_colors.items():
                legend_html += f'<p><span style="color:{color}">&#9632;</span> {species}</p>'
            legend_html += '</div>'

            map.get_root().html.add_child(folium.Element(legend_html))

            if download is True: 
                file_name = "Speciess Observation Map.html"
                map.save(file_name)
                files.download(file_name)

        return map

        """
        Plots the covariance matrix of the input dataframe as a heatmap.    
        
        Parameters:
        - env_df: pandas.DataFrame, the input dataframe
        - figsize: tuple(int, int), the figure size of the heatmap
        - text_size: int, the text size of the annotations in the heatmap

        Returns:
        - None
        """

        plt.figure(figsize=figsize)
        sns.heatmap(cov_matrix, annot=True, fmt=".2f", cmap='coolwarm', annot_kws={"size": text_size})
        plt.title('Covariance Matrix')
        plt.show()

[docs]    def boxplot_all_columns(self, df, figsize_per_plot=(10, 6), color='skyblue', fontsize=12):
        """
        Creates box plots for all numeric columns of a DataFrame.

        Args:
            df (pd.DataFrame): The DataFrame to plot.
            figsize_per_plot (tuple, optional): The figure size for each individual plot. Defaults to (4, 6).
            color (str, optional): The color of the box plots. Defaults to 'skyblue'.
            fontsize (int, optional): The fontsize of the title. Defaults to 12.

        Returns:
            None: Displays the box plots.

        Example:
            >>> import pandas as pd
            >>> data = {'A': [1, 2, 3, 4, 5], 'B': [2, 3, 4, 5, 6], 'C': [3, 4, 5, 6, 7]}
            >>> df = pd.DataFrame(data)
            >>> boxplot_all_columns(df)
        """
        numeric_cols = df.select_dtypes(include=['number']).columns
        total_plots = len(numeric_cols)
        total_height = total_plots * figsize_per_plot[1]
        fig, axs = plt.subplots(total_plots, 1, figsize=(figsize_per_plot[0], total_height))

        if total_plots == 1:
            axs = [axs]

        for i, col in enumerate(numeric_cols):
            sns.boxplot(data=df, x=col, color=color, ax=axs[i])
            axs[i].set_title(col, fontsize=fontsize)
            axs[i].set_xlabel('')
        plt.tight_layout()
        plt.show()

[docs]def plot_gdf(self, gdf, basemap=ctx.providers.OpenStreetMap.Mapnik):

  # Convert CRS to Web Mercator (epsg=3857) for compatibility with contextily basemaps
  gdf = gdf.to_crs(epsg=3857)

  # Plot the GeoDataFrame
  fig, ax = plt.subplots(figsize=(10, 10))
  gdf.plot(ax=ax, alpha=0.5, edgecolor='k')

  # Add basemap
  ctx.add_basemap(ax, source=basemap)

  # Adjust the plot limits to the bounds of the geodataframe
  ax.set_xlim(gdf.total_bounds[[0, 2]])
  ax.set_ylim(gdf.total_bounds[[1, 3]])

  plt.show()
